<?php
namespace App\Console\Commands;

use Illuminate\Console\Command;
use App\Models\Report;
use Illuminate\Support\Facades\Log;
use Carbon\Carbon;

class DataRetentionCommand extends Command
{
    protected $signature = 'reports:cleanup';
    protected $description = 'Menghapus data laporan yang telah >24 bulan dan status resolved';

    public function handle(): void
    {
        $threshold = Carbon::now()->subMonths(24);

        // Soft delete sebagai tahap pertama (aman untuk audit)
        $softDeleted = Report::where('status', 'resolved')
            ->where('created_at', '<', $threshold)
            ->whereNull('deleted_at')
            ->delete();

        // Hard delete data yang sudah soft delete >30 hari
        $hardDeleted = Report::onlyTrashed()
            ->where('status', 'resolved')
            ->where('deleted_at', '<', Carbon::now()->subDays(30))
            ->forceDelete();

        Log::info("DataRetention: {$softDeleted} laporan di-soft delete, {$hardDeleted} di-hard delete.");

        $this->info("✅ Cleanup selesai. Soft: {$softDeleted}, Hard: {$hardDeleted}");
    }
}

use Barryvdh\DomPDF\Facade\Pdf;

public function exportPdf(Request $request)
{
    $query = Report::with(['kabupaten', 'threatType'])
        ->where('status', '!=', 'archived')
        ->when($request->start, fn($q) => $q->whereDate('created_at', '>=', $request->start))
        ->when($request->end, fn($q) => $q->whereDate('created_at', '<=', $request->end))
        ->when($request->status, fn($q) => $q->where('status', $request->status));

    $reports = $query->latest()->get();

    $pdf = Pdf::loadView('reports.official', [
        'reports' => $reports,
        'start' => $request->start,
        'end' => $request->end,
        'status' => $request->status,
        'consent_label' => 'Sesuai persetujuan pelapor'
    ])->setPaper('a4', 'portrait');

    return $pdf->download('PinjolWatch_Export_' . now()->format('Y-m-d') . '.pdf');
}

class RbacSeeder extends Seeder
{
    public function run(): void
    {
        // Reset cache permissions
        app()[\Spatie\Permission\PermissionRegistrar::class]->forgetCachedPermissions();

        $permissions = [
            'view_reports', 'manage_reports', 'export_data',
            'view_audit_logs', 'manage_users', 'verify_reports'
        ];

        foreach ($permissions as $p) {
            Permission::firstOrCreate(['name' => $p]);
        }

        $roles = [
            'super_admin' => $permissions,
            'moderator' => ['view_reports', 'manage_reports', 'verify_reports', 'export_data'],
            'analyst' => ['view_reports', 'export_data'],
            'legal_advisor' => ['view_reports', 'export_data', 'verify_reports']
        ];

        foreach ($roles as $role => $perms) {
            $roleObj = Role::firstOrCreate(['name' => $role]);
            $roleObj->syncPermissions($perms);
        }
    }
}
