<?php
namespace App\Livewire;

use Livewire\Component;
use Livewire\WithFileUploads;
use App\Models\Report;
use App\Models\Kabupaten;

use Illuminate\Support\Str;
use Illuminate\Support\Facades\Hash;
use Illuminate\Validation\Rule;

class ReportForm extends Component
{
    use WithFileUploads;

    public $kabupaten_id;
    public $threat_type_id;
    public $app_name = '';
    public $chronology = '';
    public $evidence = []; // array of uploaded files
    public $consent_scope = 'internal_only';
    public $is_anonymous = true;
    public $ticket_id = null;
    public $success = false;

    protected $rules = [
        'kabupaten_id' => 'required|exists:kabupaten,id',
        'threat_type_id' => 'required|exists:threat_types,id',
        'app_name' => 'nullable|string|max:100',
        'chronology' => 'required|string|min:20|max:2000',
        'evidence.*' => 'file|mimes:jpeg,png,pdf,mp3,mp4|max:10240', // 10MB
        'consent_scope' => ['required', Rule::in(['internal_only', 'share_to_authorities', 'public_anonymized'])],
    ];

    public function submit()
    {
        $this->validate();

        $report = Report::create([
            'ticket_id' => $this->generateTicket(),
            'kabupaten_id' => $this->kabupaten_id,
            'threat_type_id' => $this->threat_type_id,
            'app_name' => $this->app_name,
            'chronology' => $this->chronology,
            'is_anonymous' => $this->is_anonymous,
            'consent_scope' => $this->consent_scope,
            'ip_hash' => Hash::make(request()->ip()),
        ]);

        if ($this->evidence) {
            foreach ($this->evidence as $file) {
                $encryptedPath = 'evidence/'.Str::uuid().'_'.$file->getClientOriginalName();
                $file->storeAs('private', $encryptedPath); // Simpan di storage/private
                $report->evidence()->create([
                    'original_name' => $file->getClientOriginalName(),
                    'encrypted_path' => Crypt::encrypt($encryptedPath),
                    'mime_type' => $file->getMimeType(),
                    'file_size' => $file->getSize(),
                ]);
            }
        }

        $this->ticket_id = $report->ticket_id;
        $this->success = true;
        $this->resetForm();
    }

    protected function resetForm()
    {
        $this->kabupaten_id = null;
        $this->threat_type_id = null;
        $this->app_name = '';
        $this->chronology = '';
        $this->evidence = [];
        $this->consent_scope = 'internal_only';
    }

    protected function generateTicket(): string
    {
        return strtoupper('PW-' . Str::random(6) . '-' . date('Ymd'));
    }
}

namespace App\Services;

use App\Models\Report;
use Illuminate\Support\Facades\Cache;
use Illuminate\Support\Facades\DB;

class ReportAggregationService
{
    public function getKabupatenStats(string $startDate = null, string $endDate = null, string $status = null): array
    {
        $cacheKey = "map_stats_{$startDate}_{$endDate}_{$status}";

        return Cache::remember($cacheKey, now()->addHours(2), function () use ($startDate, $endDate, $status) {
            $query = DB::table('reports')
                ->selectRaw('kabupaten_id, COUNT(*) as total, 
                            SUM(CASE WHEN status = "verified" THEN 1 ELSE 0 END) as verified')
                ->where('status', '!=', 'archived');

            if ($startDate) $query->whereDate('created_at', '>=', $startDate);
            if ($endDate) $query->whereDate('created_at', '<=', $endDate);
            if ($status) $query->where('status', $status);

            return $query->groupBy('kabupaten_id')
                ->pluck('total', 'kabupaten_id')
                ->map(fn($count, $kabId) => [
                    'kabupaten_id' => $kabId,
                    'count' => $count,
                    'intensity' => $count < 5 ? 1 : ($count < 20 ? 2 : 3) // 1=rendah, 2=sedang, 3=tinggi
                ])
                ->values()
                ->toArray();
        });
    }
}

        let map, geojsonLayer;

        function initMap() {
            map = L.map('map').setView([-2.5, 118], 5);
            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                attribution: '© OpenStreetMap'
            }).addTo(map);

        async function loadMap() {
            const start = document.getElementById('start').value;
            const end = document.getElementById('end').value;
            const status = document.getElementById('status').value;

            const params = new URLSearchParams({start, end, status});
            const res = await fetch(`/api/map-stats?${params}`);
            const stats = await res.json();

            if (!geojsonLayer) {
                // Load GeoJSON Kabupaten Indonesia (pastikan file tersedia di public/assets/kabupaten.geojson)
                geojsonLayer = L.geoJSON(null, {
                    style: styleFeature,
                    onEachFeature: onEachFeature
                }).addTo(map);
            }

            fetch('/assets/kabupaten.geojson')
                .then(r => r.json())
                .then(data => {
                    geojsonLayer.clearLayers();
                    // Mapping stats ke GeoJSON
                    data.features.forEach(f => {
                        const kabId = f.properties.id_kabupaten || f.properties.kode_bps;
                        f.properties.stats = stats.find(s => s.kabupaten_id == kabId) || {count: 0, intensity: 0};
                    });
                    geojsonLayer.addData(data);
                    map.fitBounds(geojsonLayer.getBounds());
                });
        }

        function styleFeature(feature) {
            const intensity = feature.properties.stats?.intensity || 0;
            const colors = ['#fee2e2', '#fecaca', '#ef4444'];
            return {
                fillColor: colors[intensity] || '#f3f4f6',
                weight: 1,
                opacity: 1,
                color: '#d1d5db',
                fillOpacity: 0.7
            };
        }

        function onEachFeature(feature, layer) {
            const s = feature.properties.stats || {count: 0};
            layer.bindTooltip(`<strong>${feature.properties.nama_kabupaten}</strong><br>Laporan: ${s.count}`, {sticky: true});
        }

        document.addEventListener('DOMContentLoaded', initMap);

namespace App\Http\Middleware;

use Closure;
use Illuminate\Http\Request;
use App\Models\AuditLog;
use Illuminate\Support\Facades\Auth;

class AuditLogMiddleware
{
    public function handle(Request $request, Closure $next)
    {
        $response = $next($request);

        if (Auth::check() && $this->isAdminRoute($request)) {
            AuditLog::create([
                'user_id' => Auth::id(),
                'route_name' => $request->route()?->getName() ?? 'unnamed',
                'http_method' => $request->method(),
                'url' => $request->fullUrl(),
                'ip_masked' => $this->maskIP($request->ip()),
                'payload_summary' => json_encode($this->summarizePayload($request)),
            ]);
        }

        return $response;
    }

    protected function isAdminRoute(Request $request): bool
    {
        return str_contains($request->path(), 'admin') || str_contains($request->path(), 'dashboard');
    }

    protected function maskIP(?string $ip): string
    {
        if (!$ip) return 'unknown';
        $parts = explode('.', $ip);
        return $parts[0] . '.' . $parts[1] . '.xxx.xxx';
    }

    protected function summarizePayload(Request $request): array
    {
        // Hapus field sensitif dari log
        $safe = $request->except(['password', 'token', 'evidence', 'chronology']);
        return array_map(fn($v) => is_array($v) ? '[array]' : mb_substr($v, 0, 50), $safe);
    }
}

        return $next($request);
    }
}
